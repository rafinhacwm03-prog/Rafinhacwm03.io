# Buscador de clientes em redes sociais
import requests
import pandas as pd

def buscar_clientes(palavras_chave, localizacao):
    # Configurar API (exemplo simplificado)
    api_key = "SUA_CHAVE_API"
    url = f"https://api.redesocial.com/busca?q={palavras_chave}&location={localizacao}&key={api_key}"
    
    response = requests.get(url)
    dados = response.json()
    
    # Processar resultados
    clientes_potenciais = []
    for item in dados['resultados']:
        cliente = {
            'nome': item['nome'],
            'perfil': item['link'],
            'segmento': item['categoria']
        }
        clientes_potenciais.append(cliente)
    
    return pd.DataFrame(clientes_potenciais)

# Usar a fun√ß√£o
df_clientes = buscar_clientes("restaurante", "S√£o Paulo")
df_clientes.to_csv('clientes_potenciais.csv', index=False)
print("Clientes encontrados e salvos com sucesso!")

# Analisador de concorrentes
from selenium import webdriver
from bs4 import BeautifulSoup
import time

def analisar_concorrentes(termo_busca):
    # Configurar navegador
    driver = webdriver.Chrome()
    driver.get(f"https://www.google.com/search?q={termo_busca}")
    
    # Esperar carregar
    time.sleep(3)
    
    # Extrair dados
    html = driver.page_source
    soup = BeautifulSoup(html, 'html.parser')
    
    resultados = []
    for item in soup.find_all('div', class_='g'):
        try:
            titulo = item.find('h3').text
            link = item.find('a')['href']
            descricao = item.find('span', class_='aCOpRe').text
            
            resultados.append({
                'titulo': titulo,
                'link': link,
                'descricao': descricao
            })
        except:
            continue
    
    driver.quit()
    return resultados

# Usar a fun√ß√£o
concorrentes = analisar_concorrentes("mercado+organic+SP")
for i, concorrente in enumerate(concorrentes[:5]):
    print(f"{i+1}. {concorrente['titulo']}")
    print(f"   Link: {concorrente['link']}")
    print()

# Automatizador de e-mails
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart

def enviar_email(destinatario, assunto, corpo):
    # Configurar e-mail
    remetente = "seu_email@gmail.com"
    senha = "sua_senha_app"  # Use senha de app
    
    # Criar mensagem
    msg = MIMEMultipart()
    msg['From'] = remetente
    msg['To'] = destinatario
    msg['Subject'] = assunto
    
    msg.attach(MIMEText(corpo, 'html'))
    
    # Enviar e-mail
    try:
        server = smtplib.SMTP('smtp.gmail.com', 587)
        server.starttls()
        server.login(remetente, senha)
        server.send_message(msg)
        server.quit()
        print(f"E-mail enviado para {destinatario}")
        return True
    except Exception as e:
        print(f"Erro ao enviar e-mail: {e}")
        return False
